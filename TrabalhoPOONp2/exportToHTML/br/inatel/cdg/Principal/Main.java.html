package br.inatel.cdg.Principal;

import br.inatel.cdg.Estoque.Estoque;
import br.inatel.cdg.Excecoes.LivroNaoEncontradoException;
import br.inatel.cdg.Threads.EstoqueThread;
import br.inatel.cdg.Arquivos.ArquivoLivros;
import br.inatel.cdg.Excecoes.QuantidadeInvalidaException;
import br.inatel.cdg.Livros.Livro;

public class Main {

    public static void main(String[] args) {

        Estoque estoque = new Estoque();

        CadastroMateriais.carregarLivros(estoque);
        CadastroMateriais.carregarArtigos(estoque);

        estoque.listarLivros();
        estoque.listarArtigos();

        EstoqueThread thread = new EstoqueThread(estoque);

        thread.start();

        ArquivoLivros.salvarLivros(estoque.getLivros());

        // Teste para ver se a exceçao funciona (Pode apagar depois)
        try {

            estoque.buscarLivro(999);

        } catch (LivroNaoEncontradoException e) {

            System.out.println(e.getMessage());
        }

        // Parar a Thread
        try {
            Thread.sleep(15000);
            thread.interrupt();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }

        // Teste para ver se a exceçao funciona (Pode apagar depois)
        try {

            Livro livro = estoque.buscarLivro(1);
            livro.removerQuantidade(-5);

        } catch (QuantidadeInvalidaException e) {
            System.out.println(e.getMessage());
        }

    }

}
